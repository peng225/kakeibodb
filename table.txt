create table kakeibo.event(
    id int primary key auto_increment, dt date,
    money int, description varchar(32)
);

create table kakeibo.tag(id int primary key auto_increment, name varchar(16) unique);

create table kakeibo.event_to_tag(
    event_id int, tag_id int,
    foreign key (event_id) references event(id) on delete cascade,
    foreign key (tag_id) references tag(id) on delete cascade,
    primary key(event_id, tag_id)
);

create table kakeibo.credit_event(
    id int primary key auto_increment, event_id int,
    dt date, money int, description varchar(32),
    foreign key(event_id) references event(id) on delete cascade
);

create table kakeibo.credit_event_to_tag(
    credit_event_id int, tag_id int,
    foreign key (credit_event_id) references credit_event(id) on delete cascade,
    foreign key (tag_id) references tag(id) on delete cascade, primary key(credit_event_id, tag_id)
);
